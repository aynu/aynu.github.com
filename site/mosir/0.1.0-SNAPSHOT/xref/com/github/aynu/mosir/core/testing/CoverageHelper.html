6   package com.github.aynu.mosir.core.testing;
7   import java.lang.reflect.Constructor;
8   import java.lang.reflect.InvocationTargetException;
9   /**
10   * カバレッジヘルパー
11   * <dl>
12   * <dt>使用条件
13   * <dd>カバレッジ関連の操作に使用すること。
14   * </dl>
15   * @author nilcy
16   */
17  public final class CoverageHelper {
18      /** 非公開コンストラクタ */
19      private CoverageHelper() {
20      }
21      /**
22       * 非公開コンストラクタ実行
23       * <dl>
24       * <dt>使用条件
25       * <dd>カバレッジ取得するために使用すること。(注:privateクラスは使用不可)
26       * </dl>
27       * @param clazz クラス
28       */
29      public static void privateConstructor(final Class<?> clazz) {
30          try {
31              final Constructor<?> constructor = clazz.getDeclaredConstructor();
32              constructor.setAccessible(true);
33              constructor.newInstance();
34              constructor.setAccessible(false);
35          } catch (NoSuchMethodException | SecurityException | InstantiationException
36              | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
37              throw new RuntimeException(e);
38          }
39      }
40  }
